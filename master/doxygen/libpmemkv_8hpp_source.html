559 {

561  std::min(size(), other.size()));

572 /*
573  * All functions which will be called by C code must be declared as extern "C"
574  * to ensure they have C linkage. It is needed because it is possible that
575  * C and C++ functions use different calling conventions.
576  */
577 extern "C" {
578 static inline int call_get_kv_function(const char *key, size_t keybytes,
579  const char *value, size_t valuebytes, void *arg)
580 {
581  return (*reinterpret_cast<std::function<get_kv_function> *>(arg))(
582  string_view(key, keybytes), string_view(value, valuebytes));
583 }
584 
585 static inline void call_get_v_function(const char *value, size_t valuebytes, void *arg)
586 {
587  (*reinterpret_cast<std::function<get_v_function> *>(arg))(
588  string_view(value, valuebytes));
589 }

591 static inline void call_get_copy(const char *v, size_t vb, void *arg)
592 {
593  auto c = reinterpret_cast<std::pair<status, std::string *> *>(arg);
594  c->first = status::OK;
595  c->second->append(v, vb);
596 }
597 }
