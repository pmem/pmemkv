634 /*
635  * All functions which will be called by C code must be declared as extern "C"
636  * to ensure they have C linkage. It is needed because it is possible that
637  * C and C++ functions use different calling conventions.
638  */
639 extern "C" {
640 static inline int call_get_kv_function(const char *key, size_t keybytes,
641  const char *value, size_t valuebytes, void *arg)
642 {
643  return (*reinterpret_cast<std::function<get_kv_function> *>(arg))(
644  string_view(key, keybytes), string_view(value, valuebytes));
645 }

648 {

650  string_view(value, valuebytes));
651 }
