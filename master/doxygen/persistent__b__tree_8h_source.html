501  std::pair<iterator, bool> insert(pool_base &pop, const_reference entry,
502  iterator begin, iterator end)
503  {
504  assert(!full());
505 
506  iterator hint =
507  std::lower_bound(begin, end, entry.first,
508  [&](const_reference entry, const key_type &key) {
509  return entry.first < key;
510  });
511 
512  if (hint != end && hint->first == entry.first) {
513  return std::pair<iterator, bool>(hint, false);
514  }
515 
516  size_t insert_pos = get_insert_idx();
517  assert(std::none_of(
518  consistent()->idxs, consistent()->idxs + size(),
519  [insert_pos](uint64_t idx) { return insert_pos == idx; }));
520  // insert an entry to the end
521  entries[insert_pos] = entry;
522  pop.flush(&(entries[insert_pos]), sizeof(entries[insert_pos]));
523  // update tmp idxs
524  size_t position = insert_idx(pop, insert_pos, hint);
525  // update consistent
526  switch_consistent(pop);
527 
528  assert(std::is_sorted(this->begin(), this->end(),
529  [](const_reference a, const_reference b) {
530  return a.first < b.first;
531  }));
532 
533  return std::pair<iterator, bool>(iterator(this, position), true);
534  }

542  size_t insert_idx(pool_base &pop, uint64_t new_entry_idx, iterator hint)
543  {
544  size_t size = this->size();
545  leaf_entries_t *tmp = working_copy();
546  auto in_begin = consistent()->idxs;
547  auto in_end = in_begin + size;
548  auto partition_point = in_begin + std::distance(this->begin(), hint);
549  auto out_begin = tmp->idxs;
550  auto insert_pos = std::copy(in_begin, partition_point, out_begin);
551  *insert_pos = new_entry_idx;
552  std::copy(partition_point, in_end, insert_pos + 1);
553  tmp->_size = size + 1;
554 #if 0
555  pop.flush( tmp->idxs, sizeof(tmp->idxs[0])*tmp->_size );
556  pop.persist( &(tmp->_size), sizeof(tmp->_size) );
557 #else
558  pop.persist(tmp, sizeof(leaf_entries_t));
559 #endif
560  auto result = std::distance(out_begin, insert_pos);
561  assert(result >= 0);
562 
563  return static_cast<std::size_t>(result);
564  }

566  void remove_idx(pool_base &pop, iterator it)
567  {
568  size_t size = this->size();
569  leaf_entries_t *tmp = working_copy();
570  auto in_begin = consistent()->idxs;
571  auto in_end = in_begin + size;
572  auto partition_point = in_begin + std::distance(this->begin(), it);
573  auto out = tmp->idxs;
574  out = std::copy(in_begin, partition_point, out);
575  out = std::copy(partition_point + 1, in_end, out);
576  *out = *partition_point;

586  void copy_insert(const_reference entry, const_iterator first, const_iterator last)
587  {
588  assert(std::distance(first, last) >= 0);
589  assert(static_cast<std::size_t>(std::distance(first, last)) <
590  number_entrys_slots);
591

596

598  consistent()->_size =
599  static_cast<std::size_t>(std::distance(entries, d_last));
600 
601  std::iota(consistent()->idxs, consistent()->idxs + consistent()->_size,
602  0);
603  }
